SELECT *
FROM ZXOCENKA_RESHENIE
LEFT OUTER JOIN (
    SELECT /*ZXOCENKA_PERSON.ID PERSONID,*/
    	   /*ZXOCENKA_PERSON.FIO FIO,*/
    	   ZXOCENKA_PERSON.DISTRICT_ID DISTRICTID,
    	   /*ZXOCENKA_HISTORY.ID AS HISTORYID,*/
    	   
    	   ZXOCENKA_HISTORY.RESHENIE_ID AS RESHENIEID,
    	   COUNT(ZXOCENKA_HISTORY.RESHENIE_ID) AS COUNTRESHENIE

    FROM ZXOCENKA_PERSON
    LEFT JOIN ZXOCENKA_HISTORY ON ZXOCENKA_PERSON.ID = ZXOCENKA_HISTORY.PERSON_ID 
    WHERE ZXOCENKA_PERSON.DISTRICT_ID = 1
	GROUP BY ZXOCENKA_PERSON.DISTRICT_ID, ZXOCENKA_HISTORY.RESHENIE_ID) AS TB1 ON ZXOCENKA_RESHENIE.ID = TB1.RESHENIEID

    	   ZXOCENKA_PERSON.DISTRICT_ID AS DISTRICTID,
    	   /*ZXOCENKA_HISTORY.ID AS HISTORYID,*/
    	   
    	   ZXOCENKA_HISTORY.RESHENIE_ID AS RESHENIEID,
    	   IF (ZXOCENKA_HISTORY.RESHENIE_ID = 'NULL','0',COUNT(ZXOCENKA_HISTORY.RESHENIE_ID)) AS COUNTRESHENIE 

    FROM ZXOCENKA_PERSON
    LEFT JOIN ZXOCENKA_HISTORY ON ZXOCENKA_PERSON.ID = ZXOCENKA_HISTORY.PERSON_ID 
    WHERE ZXOCENKA_PERSON.DISTRICT_ID = 1
	GROUP BY ZXOCENKA_PERSON.DISTRICT_ID, ZXOCENKA_HISTORY.RESHENIE_ID) AS TB1 ON ZXOCENKA_RESHENIE.ID = TB1.RESHENIEID

SELECT 
    ZXOCENKA_DECISION.ID,
    ZXOCENKA_DECISION.CNAME,
    --ZXOCENKA_HISTORY.*,
    COUNT(ZXOCENKA_PERSON.ID)
FROM ZXOCENKA_DECISION
LEFT JOIN ZXOCENKA_HISTORY ON ZXOCENKA_DECISION.ID = ZXOCENKA_HISTORY.DECISION_ID
LEFT JOIN ZXOCENKA_PERSON ON ZXOCENKA_HISTORY.PERSON_ID = ZXOCENKA_PERSON.ID
GROUP BY ZXOCENKA_DECISION.ID, ZXOCENKA_DECISION.CNAME
ORDER BY ZXOCENKA_DECISION.ID
